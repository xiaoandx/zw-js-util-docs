# 5.手写继承

# 5.1.实现方式

寄生组合式

# 5.2.说明

其它方式的继承会在一次实例中调用两次父类的构造函数或有其它缺点

# 5.3.编码实现

/* 
实现2个类型的继承
*/

function create(proto) {
  function Fn() { }
  Fn.prototype = proto
  return new Fn()
}

export function extend(Child, Parent) {
  Child.prototype = create(Parent.prototype)
  Child.prototype.constructor = Child
}
